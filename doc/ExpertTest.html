def setup
  @expert_array = []
  3.times do
    expert = create(:expert_with_availabilities)
    profile = create(:profile, user_id: expert.id)

    create(
      :profile_industry,
      profile_id: profile.id,
      industry_id: Industry.first.id
    )
    create(
      :profile_job_function,
      profile_id: profile.id,
      job_function_id: JobFunction.first.id
    )
    create(
      :profile_skill,
      profile_id: profile.id,
      skill_id: JobFunction.first.skills.first.id
    )
    create(
      :profile_seniority_level,
      profile_id: profile.id,
      seniority_level_id: SeniorityLevel.first.id
    )

    @expert_array << expert
  end
  2.times do
    expert = create(:expert_with_availabilities)
    profile = create(:profile, user_id: expert.id)

    create(
      :profile_industry,
      profile_id: profile.id,
      industry_id: Industry.second.id
    )
    create(
      :profile_job_function,
      profile_id: profile.id,
      job_function_id: JobFunction.second.id
    )
    create(
      :profile_skill,
      profile_id: profile.id,
      skill_id: JobFunction.second.skills.first.id
    )
    create(
      :profile_seniority_level,
      profile_id: profile.id,
      seniority_level_id: SeniorityLevel.second.id
    )

    @expert_array << expert
  end
end

def test_it_find_match_by_first_available
  params_one = {
    industries: ['Accounting'],
    job_functions: ['Marketing Tools and Automation'],
    seniority_levels: ['Chief Executive Officer'],
    skills: ['Advertising'],
    first_available: false
  }

  params_two = {
    industries: ['Accounting'],
    job_functions: ['Marketing Tools and Automation'],
    seniority_levels: ['Chief Executive Officer'],
    skills: ['Advertising'],
    first_available: true
  }

  expert_two = Expert.find_match(params_two)
  assert_equal 3, expert_two.count

  expert_one = Expert.find_match(params_one)
  timeslot_one = expert_one.map(&:next_availability)
  timeslot_two = expert_two.map(&:next_availability)
  ordered_time_slot_one = sort_by_timeslot(timeslot_one)
  ordered_time_slot_two = sort_by_timeslot(timeslot_two)
  assert_equal ordered_time_slot_one, ordered_time_slot_two
end
