def self.find_available_experts_today(experts_array, day = day_of_week)
  today_experts = experts_array.select do |e|
    e.availabilities.where(day_of_week: day).present?
  end

  sorted_experts = today_experts.sort_by do |expert|
    [expert.day_schedule(day: day).first]
  end

  sorted_experts
end

def self.find_match(params)
  industry_ids = params[:industries].map do |industry_name|
    Industry.find_by_name(industry_name).try(:id)
  end

  job_function_ids = params[:job_functions].map do |job_function|
    JobFunction.find_by_name(job_function).try(:id)
  end

  seniority_level_ids = params[:seniority_levels].map do |seniority_level|
    SeniorityLevel.find_by_name(seniority_level).try(:id)
  end

  skill_ids = params[:skills].map do |skill|
    Skill.find_by_name(skill).try(:id)
  end

  industries_matches = ProfileIndustry.where(
    industry_id: industry_ids
  ).map { |x| x.profile.expert }

  job_functions_matches = ProfileJobFunction.where(
    job_function_id: job_function_ids
  ).map { |x| x.profile.expert }

  seniority_levels_matches = ProfileSeniorityLevel.where(
    seniority_level_id: seniority_level_ids
  ).map { |x| x.profile.expert }

  skills_matches = ProfileSkill.where(
    skill_id: skill_ids
  ).map { |x| x.profile.expert }

  intersection = industries_matches |
                 job_functions_matches |
                 seniority_levels_matches |
                 skills_matches

  if params[:first_available] == true
    intersection = order_by_first_available(intersection)
  end

  intersection
end

def self.order_by_first_available(experts_array)
  experts = []
  next_7_days.each do |day|
    find_available_experts_today(experts_array, day).each do |expert|
      experts << expert unless experts.include? expert
    end
  end
  experts
end

def day_schedule(day: day_of_week, from_now: false)
  schedule = []
  availabilities.where(day_of_week: day).each do |availability|
    schedule << availability.start_time
  end

  if from_now == true
    next_time = closest_next_time(schedule, Time.now.strftime('%H:%M'))
    if next_time
      schedule.slice!(0, schedule.index(next_time))
      schedule
    end
  else
    schedule
  end
end

def next_availability
  first_availability = {}
  next_7_days_schedule.each do |day, _time|
    next if next_7_days_schedule[day].empty?
    next if first_availability.present?
    first_availability[day] = closest_next_time(
      next_7_days_schedule[day], Time.now.strftime('%H:%M')
    )
  end
  first_availability
end

def unavailable_times(day = day_of_week)
  schedule = base_times
  availabilities.where(day_of_week: day).each do |availability|
    schedule.delete_at(
      base_times.index(availability.start_time)
    )
  end
  schedule
end
